\noindent\textbf{Quantified Material Consequence (QMC):}\label{QMC}
\begin{equation}
\Gamma, \varphi \qmc{W} \psi \Longleftrightarrow_{df}
\begin{cases}\nonumber
1.\,\, W\subseteq\mathcal{P}(\mathcal{L}) & $ and $ \\[3pt] 
2.\,\, \forall\Delta\in W(\Gamma, \varphi,  \Delta \nmc \psi ) 
\end{cases}
\end{equation}

\noindent\textbf{Modally Robust Consequence (MRC):}\label{MRC}
\begin{equation}
\Gamma, \varphi \mrc{W} \psi \Longleftrightarrow_{df}
\begin{cases}\nonumber
1.\,\, \Gamma, \varphi \qmc{W} \psi& $ and $ \\[3pt] 
2.\,\, \forall W'(\Gamma, \varphi \qmc{W'} \psi \Longrightarrow W' \subseteq W)  & $ and $\\[3pt] 
%3.\,\, \nqmc[]{\Gamma}{B}{W} & $ and $\\[3pt]  
3.\,\, \forall\Delta(\Gamma, \varphi, \Delta \mqmc \bigperp \Longrightarrow \Delta \not\in W)

\end{cases}
\end{equation}





\noindent\textbf{Sturdy  Consequence (SC):}\label{SC}
\begin{equation}
	\Gamma, \,\varphi \smc \psi \Longrightarrow
	\begin{cases}\nonumber 
		1.\,\, \Gamma, \, \varphi \mrc{W} \psi & $ where $ \varphi \neq \psi $, and $ \\[3pt] 
		2.\,\, \forall \vartheta\,(\Gamma, \vartheta \mrc{W'} \psi \Longrightarrow W\not\subset W' ) & $ where $ \psi \neq \vartheta  \\[3pt] 
	\end{cases}
\end{equation}

\noindent\textbf{Sturdy  Consequence (SC): [Sets]}\label{SCset}
\begin{equation}
\Gamma, \, \Sigma \smc \varphi \Longrightarrow
\begin{cases}\nonumber 
1.\,\, \Gamma, \, \Sigma \mrc{W} \varphi & $ where $ \varphi \not\in \Sigma $, and $ \\[3pt] 
2.\,\, \forall \Theta\,(\Gamma, \Theta \mrc{W'} \varphi \Longrightarrow W\not\subset W' ) & $ where $ \varphi\not\in \Theta  \\[3pt] 
\end{cases}
\end{equation}
